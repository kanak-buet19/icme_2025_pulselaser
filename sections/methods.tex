\section{Methodology}

\begin{frame}{Mathematical modeling}
    
    \footnotesize 


    \begin{tcolorbox}[colback=blue!10, colframe=blue!70!black, arc=2mm, boxrule=0.5pt, top=2pt, bottom=2pt, left=2pt, right=2pt]
        \centering \textbf{Thermo-fluid Transport Equation}
    \end{tcolorbox}

    \vspace{-0.2cm}
    \[
        \nabla \cdot (\mathbf{u}) = 0
    \]
    \vspace{-0.3cm}
    \[
        \rho \left[ \frac{\partial \mathbf{u}}{\partial t} + \nabla \cdot (\mathbf{u}\mathbf{u}) \right] = \nabla \cdot (\mu \nabla \mathbf{u}) - \nabla p + \rho \beta (T - T_0) \mathbf{g} - \mu \frac{180}{\lambda^2} \frac{f_s^2}{(1 - f_s)^3} \mathbf{u}
    \]
    \vspace{-0.1cm}
    \[
        \rho c_p \left[ \frac{\partial T}{\partial t} + \nabla \cdot (\mathbf{u}T) \right] = \nabla \cdot (k \nabla T) + \rho L_f \frac{\partial f_s}{\partial t} + Q
    \]

    \begin{tcolorbox}[colback=blue!10, colframe=blue!70!black, arc=2mm, boxrule=0.5pt, top=2pt, bottom=2pt, left=2pt, right=2pt]
        \centering \textbf{Gaussian Heat Source}
    \end{tcolorbox}

    \vspace{-0.1cm}
    \[
        Q(x, y, z) = \frac{2\eta P}{r_{xy}^2 d \pi^{3/2}} \exp \left( - \left( \frac{2(x-x_0)^2}{r_{xy}^2} + \frac{2(y-y_0)^2}{r_{xy}^2} + \frac{2(z-z_0)^2}{d^2} \right) \right)
    \]

    \begin{tcolorbox}[colback=blue!10, colframe=blue!70!black, arc=2mm, boxrule=0.5pt, top=2pt, bottom=2pt, left=2pt, right=2pt]
        \centering \textbf{Heterogeneous Nucleation}
    \end{tcolorbox}

    \vspace{-0.1cm}
    \[
        f(\Delta T) = \frac{1}{\Delta T_{\sigma} \sqrt{2\pi}} \exp \left( -\frac{1}{2} \left( \frac{\Delta T_{\mu} - \Delta T}{\Delta T_{\sigma}} \right)^2 \right)
    \]

\end{frame}
